# Story 13.2: Vendors Hub Page Layout

## User Story

**As a** homeowner
**I want** a central vendors page showing all HOA service providers
**So that** I can easily navigate to detailed information about each vendor

## Acceptance Criteria

- [ ] `/vendors` route created and accessible
- [ ] Hub page displays cards for all three vendors
- [ ] Each card shows vendor name, type, tagline, and key facts
- [ ] Visual indicators for reputation concerns (BBB rating for booting company)
- [ ] Cards link to individual vendor detail pages
- [ ] Disclaimer about unofficial nature of information
- [ ] Mobile responsive layout
- [ ] Dark mode compatible

### Vendor Card Component

```typescript
// src/app/vendors/components/vendor-card.tsx
import { Card, CardHeader, CardTitle, CardDescription, CardContent, CardFooter } from "@/components/ui/card";
import { Badge } from "@/components/ui/badge";
import { Button } from "@/components/ui/button";
import { Building2, Scale, Car, AlertTriangle, Award, ArrowRight } from "lucide-react";
import Link from "next/link";
import type { VendorProfile } from "@/lib/data/vendors";

const typeIcons = {
  "property-management": Building2,
  legal: Scale,
  enforcement: Car,
};

const typeLabels = {
  "property-management": "Property Management",
  legal: "Legal Counsel",
  enforcement: "Parking Enforcement",
};

interface VendorCardProps {
  vendor: VendorProfile;
}

export function VendorCard({ vendor }: VendorCardProps) {
  const Icon = typeIcons[vendor.type];
  const hasConcerns = vendor.concerns && vendor.concerns.length > 0;
  const hasAwards = vendor.awards && vendor.awards.length > 0;

  // Find BBB rating if exists
  const bbbRating = vendor.ratings?.find((r) => r.source === "BBB");

  return (
    <Card className={hasConcerns ? "border-yellow-500/50" : ""}>
      <CardHeader>
        <div className="flex items-start justify-between">
          <div className="flex items-center gap-2">
            <Icon className="h-5 w-5 text-muted-foreground" />
            <Badge variant="secondary">{typeLabels[vendor.type]}</Badge>
          </div>
          {bbbRating && bbbRating.rating === "F" && (
            <Badge variant="destructive" className="flex items-center gap-1">
              <AlertTriangle className="h-3 w-3" />
              BBB: F
            </Badge>
          )}
          {hasAwards && !hasConcerns && (
            <Award className="h-5 w-5 text-yellow-500" />
          )}
        </div>
        <CardTitle className="mt-2">{vendor.name}</CardTitle>
        <CardDescription>{vendor.tagline}</CardDescription>
      </CardHeader>

      <CardContent>
        <p className="text-sm text-muted-foreground line-clamp-3">
          {vendor.description}
        </p>

        <div className="mt-4 flex flex-wrap gap-2 text-xs text-muted-foreground">
          <span>Since {vendor.founded}</span>
          <span>â€¢</span>
          <span>{vendor.headquarters}</span>
        </div>
      </CardContent>

      <CardFooter>
        <Button asChild className="w-full">
          <Link href={`/vendors/${vendor.slug}`}>
            Learn More <ArrowRight className="ml-2 h-4 w-4" />
          </Link>
        </Button>
      </CardFooter>
    </Card>
  );
}
```
