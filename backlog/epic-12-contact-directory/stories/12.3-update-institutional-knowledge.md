## Technical Details

### New Types

```typescript
// src/types/contacts.ts
export interface EmailContact {
  address: string;
  purpose: string;
}

export interface PhoneContact {
  number: string;
  purpose: string;
}

export interface KeyContact {
  name: string;
  title: string;
  email?: string;
  phone?: string;
  asOf?: string;
}

export interface Website {
  url: string;
  label: string;
}

export interface ContactEntityFull {
  name: string;
  role: string;
  type: "management" | "insurance" | "legal" | "board";
  emails?: EmailContact[];
  phones?: PhoneContact[];
  faxes?: PhoneContact[];
  address?: Address;
  website?: Website;
  keyContacts?: KeyContact[];
  notes?: string[];
  responsibilities?: string[];
}
```
