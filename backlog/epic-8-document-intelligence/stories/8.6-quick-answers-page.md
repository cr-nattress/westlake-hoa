# Story 8.6: Quick Answers Page

## User Story

**As a** resident looking for quick information
**I want** a page with instant answers organized by category
**So that** I can find common information without searching documents

## Description

Create a new `/answers` page that displays Quick Answer cards organized by category. Users can filter by category and search for specific topics.

## Acceptance Criteria

- [ ] Page accessible at `/answers`
- [ ] All topics from topic index displayed
- [ ] Category tabs/filters (Financial, Insurance, Rules, Governance)
- [ ] Search box to filter topics by keyword
- [ ] Responsive grid layout
- [ ] "All" option to show all categories
- [ ] Results count displayed
- [ ] Empty state when no matches
- [ ] Link in navigation

## Technical Details

### File Location
`src/app/answers/page.tsx`

### Implementation

```tsx
import { Suspense } from 'react';
import { HelpCircle, Search } from 'lucide-react';
import { Input } from '@/components/ui/input';
import { Tabs, TabsList, TabsTrigger, TabsContent } from '@/components/ui/tabs';
import { QuickAnswerCard } from '@/components/quick-answer-card';
import {
  TOPIC_INDEX,
  getTopicsByCategory,
  findTopicsByKeyword,
  getAllCategories,
  type TopicCategory,
} from '@/lib/data/topic-index';

interface AnswersPageProps {
  searchParams: { category?: string; q?: string };
}

export default function AnswersPage({ searchParams }: AnswersPageProps) {
  const { category, q } = searchParams;

  // Filter topics
  let topics = TOPIC_INDEX;

  if (q) {
    topics = findTopicsByKeyword(q);
  }

  if (category && category !== 'all') {
    topics = topics.filter((t) => t.category === category);
  }

  const categories = getAllCategories();

  return (
    <div className="container mx-auto px-4 py-8">
      {/* Header */}
      <div className="mb-8">
        <div className="flex items-center gap-3 mb-2">
          <div className="w-12 h-12 rounded-full bg-primary/10 flex items-center justify-center">
            <HelpCircle className="h-6 w-6 text-primary" />
          </div>
          <div>
            <h1 className="text-3xl font-bold">Quick Answers</h1>
            <p className="text-muted-foreground">
              Common questions answered instantly
            </p>
          </div>
        </div>
      </div>

      {/* Search */}
      <div className="mb-6">
        <form className="relative max-w-md">
          <Search className="absolute left-3 top-1/2 -translate-y-1/2 h-4 w-4 text-muted-foreground" />
          <Input
            name="q"
            placeholder="Search questions..."
            defaultValue={q}
            className="pl-10"
          />
        </form>
      </div>

      {/* Category Tabs */}
      <Tabs defaultValue={category || 'all'} className="mb-8">
        <TabsList className="flex-wrap h-auto gap-2">
          <TabsTrigger value="all" asChild>
            <a href="/answers">All</a>
          </TabsTrigger>
          {categories.map((cat) => (
            <TabsTrigger key={cat} value={cat} asChild>
              <a href={`/answers?category=${cat}`} className="capitalize">
                {cat}
              </a>
            </TabsTrigger>
          ))}
        </TabsList>
      </Tabs>

      {/* Results Count */}
      <p className="text-sm text-muted-foreground mb-4">
        {topics.length} {topics.length === 1 ? 'answer' : 'answers'} found
        {q && ` for "${q}"`}
        {category && category !== 'all' && ` in ${category}`}
      </p>

      {/* Topic Grid */}
      {topics.length > 0 ? (
        <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-4">
          {topics.map((topic) => (
            <QuickAnswerCard key={topic.slug} topic={topic} />
          ))}
        </div>
      ) : (
        <div className="text-center py-12">
          <HelpCircle className="h-12 w-12 text-muted-foreground mx-auto mb-4" />
          <h2 className="text-lg font-medium mb-2">No answers found</h2>
          <p className="text-muted-foreground mb-4">
            Try a different search term or category
          </p>
          <a href="/answers" className="text-primary hover:underline">
            View all answers
          </a>
        </div>
      )}

      {/* CTA to AI */}
      <div className="mt-12 p-6 bg-muted/50 rounded-lg text-center">
        <h2 className="font-semibold mb-2">Can&apos;t find your answer?</h2>
        <p className="text-muted-foreground mb-4">
          Ask our AI assistant for help with any HOA question
        </p>
        <a
          href="/ask"
          className="inline-flex items-center gap-2 text-primary hover:underline"
        >
          Go to AI Assistant â†’
        </a>
      </div>
    </div>
  );
}

export const metadata = {
  title: 'Quick Answers | Westlake HOA Hub',
  description: 'Find instant answers to common HOA questions about fees, insurance, rules, and governance.',
};
```
